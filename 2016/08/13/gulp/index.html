gulp

淘宝npm镜像

1.通过config命令
npm config set registry https://registry.npm.taobao.org
npm info underscore
(如果上面配置正确这个命令会有字符串response)

  1. 命令行指定
    npm --registry https://registry.npm.taobao.org info underscore
  2. 编辑 ~/.npmrc 加入下面内容
    registry = https://registry.npm.taobao.org

1.安装gulp
npm install gulp -g (global环境)
npm install gulp --save-dev (项目环境)
2.安装gulp插件
npm install gulp-minify-css gulp-concat gulp-uglify gulp-rename del
3.在项目的根目录新建gulpfile.js
` var gulp = require(‘gulp’),
jshint = require(‘gulp-jshint’),
minifycss = require(‘gulp-minify-css’),
uglify = require(‘gulp-uglify’),
concat = require(‘gulp-concat’)
del = require(‘del’);

 var jsPath = 'static/js/*.js';
 var jsBuildPath = 'dest/static/js';
 var cssPath = 'static/css/*.css';
 var cssBuildPath = 'dest/static/css';
 var distPath = 'dest/**/*';

//压缩js
gulp.task('jsCompress', function () {
    console.log('##task## js compress');
    return gulp.src(jsPath)
      .pipe(jshint())
      // .pipe(jshint.reporter('default'))
      .pipe(uglify())
      .pipe(gulp.dest(jsBuildPath));
});

//压缩css
gulp.task('cssCompress', function () {
    console.log('##task## css compress');
    return gulp.src(cssPath)
      .pipe(minifycss())
      .pipe(gulp.dest(cssBuildPath));
});

//监视jss,css文件
gulp.task('watch',function(){
    console.log('##task## watch');
    gulp.watch(jsPath, ['jsCompress','cssCompress']);
});

//清理dist文件夹
gulp.task('clean',function(cb){
    console.log('##task## clean');
    return del([distPath],cb);
});

// gulp.task('default', ['jsCompress', 'cssCompress', 'watch']);
gulp.task('default',  ['clean'], function(){
    gulp.start('jsCompress', 'cssCompress', 'watch');
});`
